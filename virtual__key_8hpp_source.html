254 constexpr bool is_control_key(vk const &key)
255 {
256  constexpr vk control_set_begin = vk::nul;
257  constexpr vk control_set_end = vk::us;
258 
259  constexpr vk abstract_set_begin = vk::del;
260  constexpr vk abstract_set_end = vk::f12;
261 
262  return (key >= control_set_begin && key <= control_set_end)
263  || (key >= abstract_set_begin && key <= abstract_set_end);
264 }
