136 constexpr character_set const* lookup_character_set(bytes code)
137 {
138  const auto len = code.size();
139 
140  if (len == 0)
141  {
142  return {};
143  }
144 
145  if (code[0] == ansi::charset_extender)
146  {
147  if (len > 1)
148  {
149  for (auto &&mapping : detail::extended_charset_map)
150  {
151  if (code[1] == mapping.second[1])
152  {
153  return &mapping.first;
154  }
155  }
156  }
157  }
158  else
159  {
160  for (auto &&mapping : detail::charset_map)
161  {
162  if (code[0] == mapping.second[0])
163  {
164  return &mapping.first;
165  }
166  }
167  }
168 
169  return nullptr;
170 }

176 constexpr boost::basic_string_view<byte> encode_character_set(character_set const &set)
177 {
178  // std::find_if is not constexpr in C++14.
179  auto charset_entry = std::cbegin(detail::charset_map);
180  while (charset_entry != std::cend(detail::charset_map))
181  {
182  if (charset_entry->first == set)
183  {
184  break;
185  }
186 
187  ++charset_entry;
188  }
189 
190  if (charset_entry != std::cend(detail::charset_map))
191  {
192  return {charset_entry->second, 1};
193  }
194  else
195  {
196  auto extended_charset_entry = std::cbegin(detail::extended_charset_map);
197  while (extended_charset_entry != std::cend(detail::extended_charset_map))
198  {
199  if (extended_charset_entry->first == set)
200  {
201  break;
202  }
203 
204  ++extended_charset_entry;
205  }
206 
207  if (extended_charset_entry != std::cend(detail::extended_charset_map))
208  {
209  return {extended_charset_entry->second, 2};
210  }
211  else
212  {
213  // If the character set is an unknown, fall back to US ASCII.
214  return encode_character_set(character_set(charset::us_ascii));
215  }
216  }
